00231 // Converting a QVariantList serialized by QJSON into a list of entities.
00232 template <class T>
00233 QList<T> fillWithList(QVariantList entities) {
00234         QList<T> res;
00235 
00236         for (QVariantList::Iterator it = entities.begin();
00237                  it != entities.end();
00238                  ++it) {
00239                 QVariant v = *it;
00240                 T entity = qVariantValue(v);
00241                 res.append(entity);
00242         }
00243 
00244         return res;
00245 }
