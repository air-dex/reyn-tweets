00827   twttr.txt.convertUnicodeIndices = function(text, entities, indicesInUTF16) {
00828         if (entities.length == 0) {
00829           return;
00830         }
00831 
00832         var charIndex = 0;
00833         var codePointIndex = 0;
00834 
00835         // sort entities by start index
00836         entities.sort(function(a,b){ return a.indices[0] - b.indices[0]; });
00837         var entityIndex = 0;
00838         var entity = entities[0];
00839 
00840         while (charIndex < text.length) {
00841           if (entity.indices[0] == (indicesInUTF16 ? charIndex : codePointIndex)) {
00842                 var len = entity.indices[1] - entity.indices[0];
00843                 entity.indices[0] = indicesInUTF16 ? codePointIndex : charIndex;
00844                 entity.indices[1] = entity.indices[0] + len;
00845 
00846                 entityIndex++;
00847                 if (entityIndex == entities.length) {
00848                   // no more entity
00849                   break;
00850                 }
00851                 entity = entities[entityIndex];
00852           }
00853 
00854           var c = text.charCodeAt(charIndex);
00855           if (0xD800 <= c && c <= 0xDBFF && charIndex < text.length - 1) {
00856                 // Found high surrogate char
00857                 c = text.charCodeAt(charIndex + 1);
00858                 if (0xDC00 <= c && c <= 0xDFFF) {
00859                   // Found surrogate pair
00860                   charIndex++;
00861                 }
00862           }
00863           codePointIndex++;
00864           charIndex++;
00865         }
