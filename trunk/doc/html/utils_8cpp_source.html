00108 QString hmacSha1(QByteArray key, QByteArray baseString) {
00109         int blockSize = 64; // HMAC-SHA-1 block size, defined in SHA-1 standard
00110 
00111         if (key.length() > blockSize) {
00112                 // if key is longer than block size (64), reduce key length with
00113                 // SHA-1 compression
00114                 key = QCryptographicHash::hash(key, QCryptographicHash::Sha1);
00115         }
00116

00118         QByteArray innerPadding(blockSize, char(0x36));
00119         // initialize outer padding with char "\"
00120         QByteArray outerPadding(blockSize, char(0x5c));
00121 
00122         // ascii characters 0x36 ("6") and 0x5c ("\") are selected because they have
00123         // large Hamming distance (http://en.wikipedia.org/wiki/Hamming_distance)
00124 
00125         for (int i = 0; i < key.length(); i++) {
00126                 // XOR operation between every byte in key and innerpadding, of key length
00127                 innerPadding[i] = innerPadding[i] ^ key.at(i);
00128                 // XOR operation between every byte in key and outerpadding, of key length
00129                 outerPadding[i] = outerPadding[i] ^ key.at(i);
00130         }
00131 
00132         // result = hash ( outerPadding CONCAT hash ( innerPadding CONCAT baseString ) ).toBase64
00133         QByteArray total = outerPadding;
00134         QByteArray part = innerPadding;
00135         part.append(baseString);
00136         total.append(QCryptographicHash::hash(part, QCryptographicHash::Sha1));
00137         QByteArray hashed = QCryptographicHash::hash(total, QCryptographicHash::Sha1);
