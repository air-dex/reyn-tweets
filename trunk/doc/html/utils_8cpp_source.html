00028 // HMAC-SHA1 algorithm for signatures.
00029 QString Utils::hmacSha1(QByteArray key, QByteArray baseString) {
00030         int blockSize = 64; // HMAC-SHA-1 block size, defined in SHA-1 standard
00031 
00032         if (key.length() > blockSize) {
00033                 // if key is longer than block size (64), reduce key length with
00034                 // SHA-1 compression
00035                 key = QCryptographicHash::hash(key, QCryptographicHash::Sha1);
00036         }
00037 
00038         // initialize inner padding with char "6"
00039         QByteArray innerPadding(blockSize, char(0x36));
00040         // initialize outer padding with char "\"
00041         QByteArray outerPadding(blockSize, char(0x5c));
00042 
00043         // ascii characters 0x36 ("6") and 0x5c ("\") are selected because they have
00044         // large Hamming distance (http://en.wikipedia.org/wiki/Hamming_distance)
00045 
00046         for (int i = 0; i < key.length(); i++) {
00047                 // XOR operation between every byte in key and innerpadding, of key length
00048                 innerPadding[i] = innerPadding[i] ^ key.at(i);
00049                 // XOR operation between every byte in key and outerpadding, of key length
00050                 outerPadding[i] = outerPadding[i] ^ key.at(i);
00051         }
00052 
00053         // result = hash ( outerPadding CONCAT hash ( innerPadding CONCAT baseString ) ).toBase64
00054         QByteArray total = outerPadding;
00055         QByteArray part = innerPadding;
00056         part.append(baseString);
00057         total.append(QCryptographicHash::hash(part, QCryptographicHash::Sha1));
00058         QByteArray hashed = QCryptographicHash::hash(total, QCryptographicHash::Sha1);
00059         return hashed.toBase64();
00060 }
