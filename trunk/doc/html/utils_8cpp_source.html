00032 // HMAC-SHA1 algorithm for signatures.
00033 QString hmacSha1(QByteArray key, QByteArray baseString) {
00034         int blockSize = 64; // HMAC-SHA-1 block size, defined in SHA-1 standard
00035 
00036         if (key.length() > blockSize) {
00037                 // if key is longer than block size (64), reduce key length with
00038                 // SHA-1 compression
00039                 key = QCryptographicHash::hash(key, QCryptographicHash::Sha1);
00040         }
00041 
00042         // initialize inner padding with char "6"
00043         QByteArray innerPadding(blockSize, char(0x36));
00044         // initialize outer padding with char "\"
00045         QByteArray outerPadding(blockSize, char(0x5c));
00046 
00047         // ascii characters 0x36 ("6") and 0x5c ("\") are selected because they have
00048         // large Hamming distance (http://en.wikipedia.org/wiki/Hamming_distance)
00049 
00050         for (int i = 0; i < key.length(); i++) {
00051                 // XOR operation between every byte in key and innerpadding, of key length
00052                 innerPadding[i] = innerPadding[i] ^ key.at(i);
00053                 // XOR operation between every byte in key and outerpadding, of key length
00054                 outerPadding[i] = outerPadding[i] ^ key.at(i);
00055         }
00056 
00057         // result = hash ( outerPadding CONCAT hash ( innerPadding CONCAT baseString ) ).toBase64
00058         QByteArray total = outerPadding;
00059         QByteArray part = innerPadding;
00060         part.append(baseString);
00061         total.append(QCryptographicHash::hash(part, QCryptographicHash::Sha1));
00062         QByteArray hashed = QCryptographicHash::hash(total, QCryptographicHash::Sha1);
00063         return hashed.toBase64();
00064 }
